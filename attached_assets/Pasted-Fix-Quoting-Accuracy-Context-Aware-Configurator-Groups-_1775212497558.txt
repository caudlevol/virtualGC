Now update the getConfiguratorOptions function (around line 639) to filter groups based on context. Replace the existing function with this:
TypeScript
export function getConfiguratorOptions(renovationType: string, conversationContext?: string) {
  const config = CONFIGURATOR_MAP[renovationType];
  if (!config) return null;

  const ctx = (conversationContext || "").toLowerCase();

  const filteredGroups = config.groups.filter((g: any) => {
    // If the group has no contextKeywords, always show it
    if (!g.contextKeywords || g.contextKeywords.length === 0) return true;
    // If we have no conversation context, show all groups
    if (!ctx) return true;
    // Only show this group if at least one of its context keywords appears in the conversation
    return g.contextKeywords.some((kw: string) => ctx.includes(kw));
  });

  // If context filtering removed ALL groups (edge case), fall back to showing all groups
  const groupsToShow = filteredGroups.length > 0 ? filteredGroups : config.groups;

  return {
    renovationType,
    label: config.label,
    groups: groupsToShow.map((g: any) => ({
      label: g.label,
      key: g.key,
      options: g.options.map((o: any) => ({
        label: o.label,
        price: o.price,
      })),
    })),
  };
}
